GeoJSON Tutorial

What this tutorial demonstrates:

This tutorial shows how to work with GeoJSON data in Leaflet maps, including different geometry types and styling options.

  • Point Features: Baseball stadiums (Coors Field, Busch Field) as point markers
  • LineString Features: Two colored line features showing how to draw paths
  • Polygon Features: State boundaries colored by political party (Republican=red, Democrat=blue)
  • Custom Styling: Different colors, weights, and opacities for various features
  • Interactive Popups: Click on features to see popup content with details
  • Conditional Rendering: Features filtered by properties (show_on_map)
  • Custom Markers: Circle markers with custom styling instead of default pins
  • Data-Driven Styling: Colors based on feature properties (political affiliation)

Note: This tutorial requires the 'map' variable to be defined. Basic map initialization is included below.

JavaScript Code (geojsonTutorial.js)

// Create GeoJSON feature, set as point features representing stadium. var geojsonFeature = { "type": "Feature", "properties": { "name": "Coors Field", "amenity": "Baseball Stadium", "popupContent": "This is where the Rockies play!" }, "geometry": { "type": "Point", "coordinates": [-104.99404, 39.75621] } }; // Add GeoJSON object to map. L.geoJSON(geojsonFeature).addTo(map); // Create two line feature objects, set their vertex coordinates. var myLines = [{ "type": "LineString", "coordinates": [[-100, 40], [-105, 45], [-110, 55]] }, { "type": "LineString", "coordinates": [[-105, 40], [-110, 45], [-115, 55]] }]; // Create myStyle variable, set color, weight, & opacity. var myStyle = { "color": "#ff7800", "weight": 5, "opacity": 0.65 }; // Set myLines object to use style settings from myStyle variable. L.geoJSON(myLines, { style: myStyle }).addTo(map); // Create states polygon features, set properties and geometry. var states = [{ "type": "Feature", "properties": {"party": "Republican"}, "geometry": { "type": "Polygon", "coordinates": [[ [-104.05, 48.99], [-97.22, 48.98], [-96.58, 45.94], [-104.03, 45.94], [-104.05, 48.99] ]] } }, { "type": "Feature", "properties": {"party": "Democrat"}, "geometry": { "type": "Polygon", "coordinates": [[ [-109.05, 41.00], [-102.06, 40.99], [-102.03, 36.99], [-109.04, 36.99], [-109.05, 41.00] ]] } }]; // Set states polygon style with fill color based on properties. Update map. L.geoJSON(states, { style: function(feature) { switch (feature.properties.party) { case 'Republican': return {color: "#ff0000"}; case 'Democrat': return {color: "#0000ff"}; } } }).addTo(map); // If a feature has a property named popupContent, attach popup features to them when clicked. function onEachFeature(feature, layer) { if (feature.properties && feature.properties.popupContent) { layer.bindPopup(feature.properties.popupContent); } }
